# File::pfopen

Try hard to find a file

# VERSION

Version 0.01

# SYNOPSIS

    use File::pfopen 'pfopen';
    my $f = pfopen('/tmp:/var/tmp:/home/njh/tmp', 'foo', 'txt:bin' );
    $f = pfopen('/tmp:/var/tmp:/home/njh/tmp', 'foo');
